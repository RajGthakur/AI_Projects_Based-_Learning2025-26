
% Tubes (Traversal) Problem using Means–End Analysis

% Define direct connections between locations
move(a, b).
move(b, c).
move(c, d).
move(d, e).
move(e, f).
move(f, g).

% The means_end/3 predicate performs traversal
% means_end(Start, Goal, Path)

means_end(Start, Goal, Path) :-
traverse(Start, Goal, [Start], PathRev),
reverse(PathRev, Path),
write('Path found: '), write(Path), nl.

% Base case: when Start and Goal are same
traverse(Goal, Goal, Path, Path) :-
write('Reached goal: '), write(Goal), nl.

% Recursive traversal using available moves
traverse(Current, Goal, Visited, Path) :-
  move(Current, Next),
\+ member(Next, Visited),
write('Moving from '), write(Current), write(' to '), write(Next), nl,
traverse(Next, Goal, [Next|Visited], Path).
